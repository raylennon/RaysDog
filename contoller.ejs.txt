            <button id="right" class="arrow-button">&#8594;</button>
        </div>
        <div>
            <button id="down" class="arrow-button">&#8595;</button>
        </div>
        <div>
            <form id="updateForm">
                <input type="text" id="message" placeholder="Message">
            </form>
        </div>
        <div>
            <button class="toggle-button selected" id="awakeButton">Awake</button>
            <button class="toggle-button" id="asleepButton">Asleep</button>
            <button class="toggle-button" id="offButton">Off</button>
        </div>
    </div>

    <script>
        const awakeButton = document.getElementById('awakeButton');
        const asleepButton = document.getElementById('asleepButton');
        const offButton = document.getElementById('offButton');

        const toggleButtons = [awakeButton, asleepButton, offButton];
        var robot_is_awake = true;

        function selectToggleButton(selectedButton) {
            toggleButtons.forEach(button => {
                if (button === selectedButton) {
                    button.classList.add('selected');
                } else {
                    button.classList.remove('selected');
                }
            });
            mode = (selectedButton.textContent || selectedButton.innerText).toLowerCase()
            robot_is_awake = (mode === "awake");
            fetch('/updateStatus', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    mode: mode,

                })
            })
        }

        toggleButtons.forEach(button => {
            button.addEventListener('click', () => {
                selectToggleButton(button);
            });
        });

        // Handle form submission
        document.getElementById('updateForm').addEventListener('input', (e) => {
            e.preventDefault();
            if (robot_is_awake) {
                const message = document.getElementById('message').value;

                // Send HTTP POST request to update the response message
                fetch('/updateStatus', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        'message': message,
                    })
                })
                    .then(response => response.text())
                    .then(data => {
                        console.log(data);
                    })
                    .catch(error => {
                        console.error(error);
                        // Handle error case
                        alert('Error occurred while updating the response message.');
                    });
                }
        });

    </script>
